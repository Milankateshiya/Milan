Q.37 Use console.log() and escape characters to print the following pattern in JS?

Ans.

        for (let i = 1; i <= 5; i++) {
            for (let j = 1; j <= 5; j++) {
              if (j == 1) {
                str = str + i + " ";
              }
              if (j == 2) {
                str = str + "1 ";
              }
              if (j == 3) {
                str = str + i + " ";
              }
              if (j == 4) {
                str = str + i * i + " ";
              }
              if (j == 5) {
                str = str + i * i * i + "";
              }
            }
            str = str + "\n";
          }
